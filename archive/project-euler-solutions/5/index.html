{% highlight ruby %} divisible.rb https://raw.github.com/addamh/euler/master/005/divisible.rb download #!/usr/bin/env ruby require "rational" def smallest_divisor_for_range(start, stop) numbers = Range.new(start, stop).to_a numbers.inject(numbers.shift) { |lcm, n| lcm.lcm(n) } end puts smallest_divisor_for_range(1, 20) {% endhighlight %} {% highlight console %} $ time ruby divisible.rb real 0m0.023s user 0m0.019s sys 0m0.003s {% endhighlight %}
